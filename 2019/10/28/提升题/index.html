  • 宏的编写
  1. #define EVALUATION(x, y) ((x / y) * y)
  2. #define EVALUATION(x, y) ((x / y + 1) * y)
  3. #define EVALUATION(x, y) \
n = ((x / y) * y);       \
while (n & (n - 1) == 0) \
{                        \
n = n + y;               \
}
  4. #define EVALUATION(x, y) \
n = ((x / y + 1) * y);   \
while (n & (n - 1) == 0) \
{                        \
n = n - y;               \
}
